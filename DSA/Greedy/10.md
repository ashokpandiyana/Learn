            total cost using online algorithm

        bought = False

        for day in range(1, days + 1):
            if not bought and day < buy_cost:
                # Rent
                online_cost += 1
                print(f"  Day {day}: Rent ($1), total=${online_cost}")
            elif not bought:
                # Buy
                online_cost += buy_cost
                bought = True
                print(f"  Day {day}: Buy (${buy_cost}), total=${online_cost}")
            else:
                # Already bought
                print(f"  Day {day}: Using owned skis, total=${online_cost}")
        
        return online_cost
    
    def offline_optimal(days, buy_cost):
        """Offline algorithm knows future, makes optimal choice."""
        if days < buy_cost:

    """

        """
        2-approximation for vertex cover.
        
        Args:
            edges: list of (u, v) edges
        
        Returns:
            vertex cover (at most 2 Ã— OPT)
        """
        cover = set()
        remaining_edges = edges.copy()
        
        print("\nEdges:", edges)
        print("\nGreedy selection:")
        
        step = 1
        while remaining_edges:
            # Pick an uncovered edge
            u, v = remaining_edges[0]
            
            # Add both endpoints to cover
            cover.add(u)
            cover.add(v)
            print(f"\nStep {step}: Pick edge ({u}, {v})")
            print(f"  Add vertices {u} and {v} to cover")
            
            # Remove all edges incident to u or v
            edges_before = len(remaining_edges)
            remaining_edges = [
                (a, b) for (a, b) in remaining_edges
                if a not in {u, v} and b not in {u, v}
            ]
            edges_covered = edges_before - len(remaining_edges)
            print(f"  Covered {edges_covered} edges, {len(remaining_edges)} remaining")
            
            step += 1
        
        return cover
