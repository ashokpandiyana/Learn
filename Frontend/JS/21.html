// Execute max N promises concurrently async function promisePool(tasks, limit) { const results = []; const executing = []; for (const [index, task] of tasks.entries()) { const promise = task().then(result => { results[index] = result; }); executing.push(promise); if (executing.length >= limit) { await Promise.race(executing); executing.splice( executing.findIndex(p => p === promise), 1 ); } } await Promise.all(executing); return results; } // Process 100 items, max 5 concurrent const tasks = items.map(item => () => processItem(item)); await promisePool(tasks, 5);
