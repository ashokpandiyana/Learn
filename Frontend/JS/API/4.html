‚ö° 4.6 DocumentFragment for Performance

üöÄ Performance Pro Tip

DocumentFragment is a lightweight container that doesn't trigger reflows during construction. Use it for batch operations!

One appendChild(fragment) = One Reflow

1000 individual appendChild() = 1000 Reflows ‚ùå

// ‚ùå BAD: Multiple reflows for (let i = 0; i < 1000; i++) { const li = document.createElement('li'); li.textContent = `Item ${i}`; ul.appendChild(li); // 1000 reflows! üò± } // ‚úÖ GOOD: Single reflow with DocumentFragment const fragment = document.createDocumentFragment(); for (let i = 0; i < 1000; i++) { const li = document.createElement('li'); li.textContent = `Item ${i}`; fragment.appendChild(li); // No reflow } ul.appendChild(fragment); // Only 1 reflow! üöÄ

üéØ Real-World Examples

Example 1: Dynamic List Builder

function createUserList(users) { const ul = document.createElement('ul'); ul.className = 'user-list'; const fragment = document.createDocumentFragment(); users.forEach(user => { const li = document.createElement('li'); li.innerHTML = ` <img src="${user.avatar}" alt="${user.name}"> <span>${user.name}</span> `; fragment.appendChild(li); }); ul.appendChild(fragment); return ul; }

Example 2: Safe HTML Builder

function createSafeCard(title, content) { const card = document.createElement('div'); card.className = 'card'; const titleEl = document.createElement('h3'); titleEl.textContent = title; // Safe from XSS const contentEl = document.createElement('p'); contentEl.textContent = content; // Safe from XSS card.append(titleEl, contentEl); return card; }
‚úÖ Best Practice: Always use createElement + textContent when dealing with user input. Only use innerHTML with trusted, static HTML.
