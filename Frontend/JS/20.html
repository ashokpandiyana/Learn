function debounce(fn, delay) { let timeoutId; return function(...args) { const context = this; clearTimeout(timeoutId); timeoutId = setTimeout(() => { fn.apply(context, args); }, delay); }; }

2.3 Flatten Array

function flatten(arr, depth = 1) { if (depth === 0) return arr; return arr.reduce((acc, item) => { if (Array.isArray(item)) { return acc.concat(flatten(item, depth - 1)); } return acc.concat(item); }, []); }
