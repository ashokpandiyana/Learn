3.1 Type Guards and Narrowing

Type narrowing is the process of refining a type to a more specific type within a conditional block. TypeScript uses type guards to determine the narrowed type.

typeof Type Guards

The typeof operator works with primitive types:

function processValue(value: string | number | boolean) {
  // typeof narrowing
  if (typeof value === "string") {
    // TypeScript knows value is string here
    console.log(value.toUpperCase()); // ‚úÖ string methods available
  } else if (typeof value === "number") {
    // TypeScript knows value is number here
    console.log(value.toFixed(2));    // ‚úÖ number methods available
  } else {
    // TypeScript knows value is boolean here
    console.log(value ? "Yes" : "No");
  }
}

// typeof checks JavaScript can use:
// "string", "number", "boolean", "undefined", "object", "function", "symbol", "bigint"

instanceof Type Guards

The instanceof operator works with class instances:

class Dog {
  bark() { console.log("Woof!"); }
}

class Cat {
  meow() { console.log("Meow!"); }
}

function makeSound(animal: Dog | Cat) {
  if (animal instanceof Dog) {
    animal.bark();  // ‚úÖ TypeScript knows it's a Dog
  } else {
    animal.meow();  // ‚úÖ TypeScript knows it's a Cat
  }
}

// Also works with built-in types
function processDate(value: Date | string) {
  if (value instanceof Date) {
    console.log(value.getFullYear()); // Date methods available
  } else {
    console.log(value.split("-"));      // string methods available
  }
}

in Operator Narrowing

The in operator checks if a property exists in an object:

interface Bird {
  fly(): void;
  wings: number;
}

interface Fish {
  swim(): void;
  fins: number;
}

function move(animal: Bird | Fish) {
  if ("fly" in animal) {
    animal.fly();   // ‚úÖ TypeScript knows it's a Bird
    console.log(animal.wings);
  } else {
    animal.swim();  // ‚úÖ TypeScript knows it's a Fish
    console.log(animal.fins);
  }
}

Equality Narrowing

function example(x: string | number, y: string | boolean) {
  if (x === y) {
    // Both must be string (the only common type)
    console.log(x.toUpperCase()); // ‚úÖ string
    console.log(y.toUpperCase()); // ‚úÖ string
  }
}

// Narrowing with null checks
function processOptional(value: string | null | undefined) {
  if (value != null) { // Checks both null and undefined
    console.log(value.toUpperCase()); // ‚úÖ string
  }
}

Custom Type Guards (User-Defined Type Guards)

Create your own type guards using the is keyword:

interface User {
  type: "user";
  name: string;
  email: string;
}

interface Admin {
  type: "admin";
  name: string;
  permissions: string[];
}

// Custom type guard function
function isAdmin(person: User | Admin): person is Admin {
  return person.type === "admin";
}

function handlePerson(person: User | Admin) {
  if (isAdmin(person)) {
    // TypeScript knows person is Admin
    console.log(person.permissions); // ‚úÖ
  } else {
    // TypeScript knows person is User
    console.log(person.email); // ‚úÖ
  }
}

// Type guard for arrays
function isStringArray(arr: unknown): arr is string[] {
  return Array.isArray(arr) && arr.every(item => typeof item === "string");
}

// Type guard for object shape
function hasProperty<T, K extends string>(
  obj: T,
  prop: K
): obj is T & Record<K, unknown> {
  return typeof obj === "object" && obj !== null && prop in obj;
}

Assertion Functions

Assertion functions throw an error if a condition isn't met:

// Assertion function with asserts keyword
function assertIsString(value: unknown): asserts value is string {
  if (typeof value !== "string") {
    throw new Error(`Expected string, got ${typeof value}`);
  }
}

function processValue(value: unknown) {
  assertIsString(value);
  // After the assertion, TypeScript knows value is string
  console.log(value.toUpperCase()); // ‚úÖ
}

// Assert non-null
function assertDefined<T>(value: T): asserts value is NonNullable<T> {
  if (value === null || value === undefined) {
    throw new Error("Value must be defined");
  }
}

3.2 Conditional Types

Conditional types select one of two types based on a condition, similar to the ternary operator in JavaScript.

üîÄ Conditional Type Syntax
T extends U ? X : Y

// If T is assignable to U, the type is X, otherwise Y

Basic Conditional Types

// Simple conditional type
type IsString<T> = T extends string ? true : false;

type A = IsString<string>;    // true
type B = IsString<number>;    // false
type C = IsString<"hello">;   // true (literal extends string)

// Flatten array type
type Flatten<T> = T extends Array<infer U> ? U : T;

type Str = Flatten<string[]>;     // string
type Num = Flatten<number>;       // number (not an array)

// Check for null/undefined
type IsNullable<T> = T extends null | undefined ? true : false;

Distributive Conditional Types

When a conditional type acts on a union, it distributes over each member:

type ToArray<T> = T extends any ? T[] : never;

// Distribution happens!
type Result = ToArray<string | number>;
// = ToArray<string> | ToArray<number>
// = string[] | number[]

// Prevent distribution with tuple
type ToArrayNonDist<T> = [T] extends [any] ? T[] : never;

type Result2 = ToArrayNonDist<string | number>;
// = (string | number)[]  -- No distribution!

The infer Keyword

The infer keyword lets you extract types from other types within conditional types:

// Extract return type of a function
type MyReturnType<T> = T extends (...args: any[]) => infer R ? R : never;

type FnReturn = MyReturnType<() => string>;  // string
type FnReturn2 = MyReturnType<(x: number) => boolean>; // boolean

// Extract parameters
type MyParameters<T> = T extends (...args: infer P) => any ? P : never;

type Params = MyParameters<(a: string, b: number) => void>;
// [a: string, b: number]

// Extract array element type
type ElementType<T> = T extends (infer E)[] ? E : T;

type Elem = ElementType<string[]>;  // string

// Extract Promise value
type Unpromise<T> = T extends Promise<infer U> ? U : T;

type Value = Unpromise<Promise<string>>; // string

Nested Conditional Types

// Multiple conditions
type TypeName<T> =
  T extends string ? "string" :
  T extends number ? "number" :
  T extends boolean ? "boolean" :
  T extends undefined ? "undefined" :
  T extends Function ? "function" :
  "object";

type T1 = TypeName<string>;        // "string"
type T2 = TypeName<() => void>;    // "function"
type T3 = TypeName<string[]>;      // "object"

3.3 Mapped Types

Mapped types allow you to create new types by transforming properties of existing types.

üó∫Ô∏è Mapped Type Syntax
{ [K in keyof T]: TransformedType }

// Iterate over each key K in type T and transform

Basic Mapped Types

interface Person {
  name: string;
  age: number;
  email: string;
}

// Make all properties optional
type MyPartial<T> = {
  [K in keyof T]?: T[K];
};

type PartialPerson = MyPartial<Person>;
// { name?: string; age?: number; email?: string; }

// Make all properties required
type MyRequired<T> = {
  [K in keyof T]-?: T[K]; // -? removes optional
};

// Make all properties readonly
type MyReadonly<T> = {
  readonly [K in keyof T]: T[K];
};

// Remove readonly (mutable)
type Mutable<T> = {
  -readonly [K in keyof T]: T[K]; // -readonly removes readonly
};

Key Remapping with as

TypeScript 4.1+ allows renaming keys during mapping:

interface Person {
  name: string;
  age: number;
}

// Add prefix to all keys
type Getters<T> = {
  [K in keyof T as `get${Capitalize<K & string>}`]: () => T[K];
};

type PersonGetters = Getters<Person>;
// { getName: () => string; getAge: () => number; }

// Filter keys based on value type
type OnlyStrings<T> = {
  [K in keyof T as T[K] extends string ? K : never]: T[K];
};

type PersonStrings = OnlyStrings<Person>;
// { name: string; }  -- age is excluded (it's number)

// Remove specific keys
type RemoveKeys<T, K> = {
  [P in keyof T as P extends K ? never : P]: T[P];
};

type WithoutAge = RemoveKeys<Person, "age">;
// { name: string; }

Implementing Pick and Omit

// Pick: Select specific keys
type MyPick<T, K extends keyof T> = {
  [P in K]: T[P];
};

type NameOnly = MyPick<Person, "name">; // { name: string; }

// Omit: Exclude specific keys
type MyOmit<T, K extends keyof T> = {
  [P in keyof T as P extends K ? never : P]: T[P];
};

// Alternative Omit using Exclude
type MyOmit2<T, K extends keyof T> = Pick<T, Exclude<keyof T, K>>;

type WithoutName = MyOmit<Person, "name">; // { age: number; }

3.4 Template Literal Types (Advanced)

Template literal types combine string literal types with template literal syntax for powerful string manipulation at the type level.

// Event handler type generator
type EventNames = "click" | "focus" | "blur" | "change";

type EventHandlers = {
  [E in EventNames as `on${Capitalize<E>}`]: (event: Event) => void;
};
// { onClick: (event: Event) => void; onFocus: ...; onBlur: ...; onChange: ...; }

// CSS property types
type CSSProperty = "margin" | "padding";
type Direction = "Top" | "Right" | "Bottom" | "Left";

type SpacingProperties = `${CSSProperty}${Direction}`;
// "marginTop" | "marginRight" | "marginBottom" | "marginLeft" | 
// "paddingTop" | "paddingRight" | "paddingBottom" | "paddingLeft"

// Route parameters extraction
type ExtractParams<T extends string> =
  T extends `${infer _}:${infer Param}/${infer Rest}`
    ? Param | ExtractParams<Rest>
    : T extends `${infer _}:${infer Param}`
      ? Param
      : never;

type RouteParams = ExtractParams<"/users/:userId/posts/:postId">;
// "userId" | "postId"

3.5 Utility Types (Must Know!)

TypeScript provides several built-in utility types. These are essential for interviews and daily development.

Partial<T>

Makes all properties optional

Required<T>

Makes all properties required

Readonly<T>

Makes all properties readonly

Record<K, V>

Object with keys K and values V

Pick<T, K>

Select specific properties

Omit<T, K>

Remove specific properties

Exclude<T, U>

Remove types from union

Extract<T, U>

Keep matching types from union

NonNullable<T>

Remove null and undefined

ReturnType<T>

Get function return type

Parameters<T>

Get function params as tuple

Awaited<T>

Unwrap Promise type

Utility Types in Action

interface User {
  id: string;
  name: string;
  email: string;
  age?: number;
}

// Partial - for update operations
function updateUser(id: string, updates: Partial<User>) {
  // updates can have any combination of User properties
}
updateUser("1", { name: "New Name" }); // ‚úÖ

// Required - ensure all fields are present
type CompleteUser = Required<User>; // age is now required

// Pick - select specific fields
type UserCredentials = Pick<User, "email" | "name">;
// { email: string; name: string; }

// Omit - exclude fields
type UserWithoutId = Omit<User, "id">;
// { name: string; email: string; age?: number; }

// Record - typed object
type UserRoles = Record<string, "admin" | "user" | "guest">;
const roles: UserRoles = {
  "user1": "admin",
  "user2": "user"
};

// Exclude and Extract with unions
type AllTypes = string | number | boolean | null;
type NoNull = Exclude<AllTypes, null>;  // string | number | boolean
type OnlyPrimitive = Extract<AllTypes, string | number>; // string | number

// NonNullable
type MaybeString = string | null | undefined;
type DefinitelyString = NonNullable<MaybeString>; // string

// ReturnType and Parameters
function createUser(name: string, age: number): User {
  return { id: "1", name, email: "", age };
}

type CreateUserReturn = ReturnType<typeof createUser>; // User
type CreateUserParams = Parameters<typeof createUser>; // [string, number]

// Awaited - unwrap Promise
type PromiseUser = Promise<User>;
type UnwrappedUser = Awaited<PromiseUser>; // User

Production Tip: Combining Utility Types

You can combine utility types for powerful patterns:

// Make specific fields required, others optional
type RequiredPick<T, K extends keyof T> = 
  Required<Pick<T, K>> & Partial<Omit<T, K>>;

// Create form with required ID but optional other fields
type CreateUserForm = RequiredPick<User, "name" | "email">;
// name and email required, id and age optional
