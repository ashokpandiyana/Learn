6.1 Function Component Types

There are multiple ways to type React function components. Understanding the differences helps you choose the right approach for your codebase.

The React.FC Type (and Why Many Avoid It)

import React from 'react';

interface GreetingProps {
  name: string;
  age?: number;
}

// Using React.FC (Function Component)
const Greeting: React.FC<GreetingProps> = ({ name, age }) => {
  return (
    <div>
      Hello, {name}! {age && <span>Age: {age}</span>}
    </div>
  );
};
‚ö†Ô∏è

Interview Alert: Problems with React.FC

  1. Implicit children: Before React 18, FC included children even if not wanted
  2. No generic support: Can't create generic components with FC
  3. Doesn't work with defaultProps: Type inference issues
  4. Verbose: Adds extra type annotation

Preferred Approach: Direct Typing

interface GreetingProps {
  name: string;
  age?: number;
  children?: React.ReactNode; // Explicitly add if needed
}

// ‚úÖ Preferred: Type props directly
function Greeting({ name, age, children }: GreetingProps) {
  return (
    <div>
      Hello, {name}!
      {age && <span> Age: {age}</span>}
      {children}
    </div>
  );
}

// Also works with arrow functions
const Greeting2 = ({ name, age }: GreetingProps) => (
  <div>Hello, {name}!</div>
);

// With explicit return type (optional but recommended for public APIs)
function Greeting3({ name }: GreetingProps): JSX.Element {
  return <div>Hello, {name}!</div>;
}
‚ùå React.FC Issues
  • Implicit children (before React 18)
  • Can't use generics easily
  • defaultProps doesn't work well
  • More verbose
‚úÖ Direct Typing Benefits
  • Explicit about children
  • Supports generics
  • Better type inference
  • Industry standard

Generic Components

Generic components allow you to create reusable components with type-safe props:

// Generic List component
interface ListProps<T> {
  items: T[];
  renderItem: (item: T, index: number) => React.ReactNode;
  keyExtractor: (item: T) => string;
}

// Generic function component
function List<T>({ items, renderItem, keyExtractor }: ListProps<T>) {
  return (
    <ul>
      {items.map((item, index) => (
        <li key={keyExtractor(item)}>
          {renderItem(item, index)}
        </li>
      ))}
    </ul>
  );
}

// Usage with type inference
interface User {
  id: string;
  name: string;
  email: string;
}

const users: User[] = [
  { id: '1', name: 'Alice', email: 'alice@example.com' },
  { id: '2', name: 'Bob', email: 'bob@example.com' },
];

// TypeScript infers T as User
<List
  items={usersrenderItem={(user) => <span>{user.name}</span>}  // user is typed as User
  keyExtractor={(user) => user.id// Or explicitly specify the type
<List<User>
  items={users}
  renderItem={(user) => <span>{user.email}</span>}
  keyExtractor={(user) => user.id}
/>

6.2 Props Typing

Required vs Optional Props

interface ButtonProps {
  // Required props
  label: string;
  onClick: () => void;
  
  // Optional props
  variant?: 'primary' | 'secondary' | 'danger';
  size?: 'sm' | 'md' | 'lg';
  disabled?: boolean;
  icon?: React.ReactNode;
}

// Default values with destructuring
function Button({
  label,
  onClick,
  variant = 'primary',     // Default value
  size = 'md',              // Default value
  disabled = false,
  icon,
}: ButtonProps) {
  return (
    <button
      onClick={onClick}
      disabled={disabled}
      className={`btn btn-${variant} btn-${size}`}
    >
      {icon && <span className="icon">{icon}</span>}
      {label}
    </button>
  );
}

React Node Types

Understanding the different types for React children and content:

React.ReactNode

Most flexible. Accepts: elements, strings, numbers, fragments, portals, null, undefined, booleans

React.ReactElement

Only React elements (JSX). No strings, numbers, or null

JSX.Element

Return type of JSX expressions. Subset of ReactElement

React.PropsWithChildren<P>

Utility to add children prop to existing props type

import type { ReactNode, ReactElement, PropsWithChildren } from 'react';

// ReactNode - accepts almost anything renderable
interface CardProps {
  title: string;
  children: ReactNode;  // Most common for children
}

function Card({ title, children }: CardProps) {
  return (
    <div className="card">
      <h2>{title}</h2>
      {childrendiv>
  );
}

// All valid:
<Card title="Test">Hello</Card>            // string
<Card title="Test">{42}</Card>               // number
<Card title="Test"><span>Hi</span></Card>    // element
<Card title="Test">{null}</Card>             // null

// ReactElement - stricter, only elements
interface LayoutProps {
  header: ReactElement;  // Must be a React element
  sidebar: ReactElement;
  children: ReactNode;
}

// PropsWithChildren utility
type CardPropsAlt = PropsWithChildren<{
  title: string;
  subtitle?: string;
}>;
// Equivalent to: { title: string; subtitle?: string; children?: ReactNode }

ComponentProps Utility Types

Extract props from existing components or HTML elements:

import type { ComponentProps, ComponentPropsWithRef, ComponentPropsWithoutRef } from 'react';

// Get props of HTML elements
type ButtonHTMLProps = ComponentProps<'button'>;
type InputHTMLProps = ComponentProps<'input'>;
type DivHTMLProps = ComponentProps<'div'>;

// Extend HTML element props
interface CustomButtonProps extends ComponentPropsWithoutRef<'button'> {
  variant: 'primary' | 'secondary';
  isLoading?: boolean;
}

function CustomButton({ 
  variant, 
  isLoading, 
  children, 
  className,
  ...rest  // All other button props (onClick, disabled, etc.)
}: CustomButtonProps) {
  return (
    <button
      className={`btn btn-${variant} ${className || ''}`}
      disabled={isLoading}
      {...restisLoading ? 'Loading...' : children}
    </button>
  );
}

// Get props of custom component
type CustomButtonPropsFromComponent = ComponentProps<typeof CustomButton>;

// Usage - has all button props plus custom ones
<CustomButton
  variant="primary"
  onClick={() => console.log('clicked')}
  type="submit"
  disabled={false}
>
  Submit
</CustomButton>

Interview Alert: ComponentPropsWithRef vs WithoutRef

ComponentPropsWithoutRef: Use when your component doesn't forward refs. This is most common.

ComponentPropsWithRef: Use when your component forwards refs using forwardRef.

6.3 Children Types Deep Dive

import type { ReactNode, ReactElement } from 'react';

// 1. ReactNode - Most common and flexible
interface ContainerProps {
  children: ReactNode;
}

// 2. ReactElement - Only JSX elements
interface TabPanelProps {
  children: ReactElement;  // Must be single element
}

// 3. Array of elements
interface TabsProps {
  children: ReactElement[];  // Must be array of elements
}

// 4. Specific element type
interface NavProps {
  children: ReactElement<NavItemProps> | ReactElement<NavItemProps>[];
}

// 5. Function as children (Render Props pattern)
interface DataProviderProps<T> {
  url: string;
  children: (data: T, loading: boolean, error: Error | null) => ReactNode;
}

function DataProvider<T>({ url, children }: DataProviderProps<T>) {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<Error | null>(null);
  
  // Fetch logic...
  
  return <>{children(data as T, loading, error)}</>;
}

// Usage of function as children
<DataProvider<User[]> url="/api/users">
  {(users, loading, error) => (
    <div>
      {loading && <p>Loading...</p>}
      {error && <p>Error: {error.message}</p>}
      {users && users.map(u => <div key={u.id}>{u.name}</div>)}
    </div>
  )}
</DataProvider>

6.4 Event Handler Types

React provides specific event types for type-safe event handling:

Event Type Use Case
React.MouseEvent<HTMLElement> Click, double-click, mouse enter/leave
React.ChangeEvent<HTMLInputElement> Input value changes
React.FormEvent<HTMLFormElement> Form submission
React.KeyboardEvent<HTMLElement> Key press, key down, key up
React.FocusEvent<HTMLElement> Focus, blur
React.DragEvent<HTMLElement> Drag and drop 
import type { 
  MouseEvent, 
  ChangeEvent, 
  FormEvent, 
  KeyboardEvent,
  FocusEvent
} from 'react';

function EventExamples() {
  // Mouse event
  const handleClick = (e: MouseEvent<HTMLButtonElement>) => {
    console.log('Clicked at:', e.clientX, e.clientY);
    console.log('Button:', e.currentTarget.name);
  };

  // Change event - different element types
  const handleInputChange = (e: ChangeEvent<HTMLInputElement>) => {
    console.log('Input value:', e.target.value);
  };

  const handleSelectChange = (e: ChangeEvent<HTMLSelectElement>) => {
    console.log('Selected:', e.target.value);
  };

  const handleTextareaChange = (e: ChangeEvent<HTMLTextAreaElement>) => {
    console.log('Textarea:', e.target.value);
  };

  // Form submit
  const handleSubmit = (e: FormEvent<HTMLFormElement>) => {
    e.preventDefault();
    const formData = new FormData(e.currentTarget);
    console.log('Form data:', Object.fromEntries(formData));
  };

  // Keyboard event
  const handleKeyDown = (e: KeyboardEvent<HTMLInputElement>) => {
    if (e.key === 'Enter') {
      console.log('Enter pressed!');
    }
    if (e.key === 'Escape') {
      e.currentTarget.blur();
    }
  };

  // Focus event
  const handleFocus = (e: FocusEvent<HTMLInputElement>) => {
    console.log('Focused:', e.target.name);
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        name="email"
        onChange={handleInputChange}
        onKeyDown={handleKeyDown}
        onFocus={handleFocus}
      />
      <select onChange={handleSelectChange}>
        <option value="a">A</option>
        <option value="b">B</option>
      </select>
      <button name="submit" onClick={handleClick}>Submit</button>
    </form>
  );
}

6.5 Style Types

import type { CSSProperties } from 'react';

interface BoxProps {
  style?: CSSProperties;  // React's inline style type
  className?: string;
}

function Box({ style, className }: BoxProps) {
  // Default styles with type safety
  const defaultStyles: CSSProperties = {
    display: 'flex',
    flexDirection: 'column',
    padding: 16,
    borderRadius: 8,
    backgroundColor: '#f0f0f0',
    // TypeScript catches typos!
    // colour: 'red', // ‚ùå Error: 'colour' does not exist
  };

  return (
    <div 
      className={className} 
      style={{ ...defaultStyles, ...style }}
    />
  );
}

// Dynamic styles with type safety
interface StyledBoxProps {
  color: 'primary' | 'secondary' | 'danger';
  size: 'sm' | 'md' | 'lg';
}

function StyledBox({ color, size }: StyledBoxProps) {
  const colorMap: Record<StyledBoxProps['color'], string> = {
    primary: '#3b82f6',
    secondary: '#64748b',
    danger: '#ef4444',
  };

  const sizeMap: Record<StyledBoxProps['size'], number> = {
    sm: 8,
    md: 16,
    lg: 24,
  };

  const styles: CSSProperties = {
    backgroundColor: colorMap[color],
    padding: sizeMap[size],
  };

  return <div style={styles} />;
}
